Golang Practice Exercises - While Loops

Complete the following exercises to strengthen your understanding of While Loops Concepts.

1. Sum of Even Numbers

Write a loop to find the sum of all even numbers in a given list.

numbers := []int{3, 5, 8, 12, 7, 10, 14}
Solution 
package main

import "fmt"

func main() {
    numbers := []int{3, 5, 8, 12, 7, 10, 14}
    totalSum := 0
    index := 0

    for index < len(numbers) {
        if numbers[index]%2 == 0 {
            totalSum += numbers[index]
        }
        index++
    }

    fmt.Println("Sum of even numbers:", totalSum)
}

2. Find the Maximum Value

Use a loop to find the maximum value in a given list of numbers.

numbers := []int{23, 89, 12, 67, 34}
Solution 
package main

import "fmt"

func main() {
    numbers := []int{23, 89, 12, 67, 34}
    maxValue := numbers[0]
    index := 1

    for index < len(numbers) {
        if numbers[index] > maxValue {
            maxValue = numbers[index]
        }
        index++
    }

    fmt.Println("Maximum value:", maxValue)
}

4. Collatz Conjecture

Implement the Collatz Conjecture: given a number, if it's even, divide it by 2, and if it's odd, multiply it by 3 and add 1. Repeat this process until n becomes 1. 

number := 6
Solution 
package main

import "fmt"

func main() {
    n := 6
    for n != 1 {
        fmt.Print(n, " ")
        if n%2 == 0 {
            n = n / 2
        } else {
            n = n*3 + 1
        }
    }
    fmt.Println(n)
}

5. Count the Digits in a Number

Write a loop to count the number of digits in a given integer. For example, the number 7890 should result in a count of 4 digits. 

number := 7890
Solution 
package main

import "fmt"

func main() {
    number := 7890
    digitCount := 0

    for number > 0 {
        number /= 10
        digitCount++
    }

    fmt.Println("Number of digits:", digitCount)
}

6. Find the Minimum Digit in a Number

Write a loop to find the smallest digit in a given number. For example, if the number is 57321, the minimum digit should be 1. 

number := 57321
Solution 
package main

import "fmt"

func main() {
    number := 57321
    minDigit := 9

    for number > 0 {
        digit := number % 10
        if digit < minDigit {
            minDigit = digit
        }
        number /= 10
    }

    fmt.Println("Minimum digit:", minDigit)
}

7. Count the Even Digits in a Number

Write a loop to count the number of even digits in a given number. For example, if the number is 47382, the count of even digits should be 3 because 4, 8, and 2 are even. 

number := 47382
Solution

func main() {

8. Reverse a Number Using a While Loop

Write a loop to reverse the digits of a number. For example, if the number is 12345, the reversed number should be 54321. 

number := 12345
Solution 
package main

import "fmt"

func main() {
    number := 12345
    reversed := 0

    for number > 0 {
        digit := number % 10
        reversed = reversed*10 + digit
        number /= 10
    }

    fmt.Println("Reversed number:", reversed)
}

9. Check if a Number is a Palindrome

Write a loop to check if a number is a palindrome. A number is a palindrome if it reads the same forward and backward. For example, 12321 is a palindrome, but 12345 is not. 

number := 12321
Solution 
package main

import "fmt"

func main() {
    number := 12321
    original := number
    reversed := 0

    for number > 0 {
        digit := number % 10
        reversed = reversed*10 + digit
        number /= 10
    }

    if original == reversed {
        fmt.Println("The number is a palindrome.")
    } else {
        fmt.Println("The number is not a palindrome.")
    }
}

10. Sum of Digits Until a Single Digit

Write a loop to repeatedly sum the digits of a number until the result is a single digit. For example, if the number is 9876, the process is 9 + 8 + 7 + 6 = 30, then 3 + 0 = 3. The final result is 3. 

number := 9876
Solution 
package main

import "fmt"

func main() {
    number := 9876

    for number >= 10 {
        sum := 0
        for number > 0 {
            digit := number % 10
            sum += digit
            number /= 10
        }
        number = sum
    }

    fmt.Println("Single digit sum:", number)
}

11. Check if a Number is an Armstrong Number

Write a loop to check if a given number is an Armstrong number. A number is an Armstrong number if the sum of its digits each raised to the power of the number of digits equals the original number. For example, 153 is an Armstrong number because 1^3 + 5^3 + 3^3 = 153, but 123 is not. 

number := 153
Solution 
package main

import (
    "fmt"
    "math"
)

func main() {
    number := 153
    original := number
    count := 0
    sum := 0

    // Count the number of digits
    temp := number
    for temp > 0 {
        count++
        temp /= 10
    }

    // Compute the sum of digits raised to the power of count
    temp = number
    for temp > 0 {
        digit := temp % 10
        sum += int(math.Pow(float64(digit), float64(count)))
        temp /= 10
    }

    if sum == original {
        fmt.Println("The number is an Armstrong number.")
    } else {
        fmt.Println("The number is not an Armstrong number.")
    }
}
