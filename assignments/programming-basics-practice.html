4. User Input - Favorite Number

Write a program that asks the user for their favorite number, doubles it, and prints the result.

Solution 
package main
import "fmt"

func main() {
    var favoriteNumber int
    fmt.Print("Enter your favorite number: ")
    fmt.Scan(&favoriteNumber)
    doubledNumber := favoriteNumber * 2
    fmt.Printf("Double your favorite number is %d\n", doubledNumber)
}

6. Calculate the Area of a Circle

Write a program that takes a radius (number) from the user, calculates the area of a circle, and prints it.

Solution 
package main
import "fmt"

func main() {
    const pi = 3.14
    var radius float64
    fmt.Print("Enter a radius: ")
    fmt.Scan(&radius)
    area := pi * radius * radius
    fmt.Printf("The area of the circle is %.2f\n", area)
}

7. Convert Fahrenheit to Celsius

Convert a temperature from Fahrenheit to Celsius.

Solution 
package main
import "fmt"

func main() {
    var fahrenheit float64
    fmt.Print("Enter temperature in Fahrenheit: ")
    fmt.Scan(&fahrenheit)
    celsius := (fahrenheit - 32) * 5 / 9
    fmt.Printf("Temperature in Celsius is %.2f\n", celsius)
}

8. Calculate the Perimeter of a Rectangle

Write a program that takes the length and width of a rectangle, calculates the perimeter, and prints it.

Solution 
package main
import "fmt"

func main() {
    var length, width int
    fmt.Print("Enter the length of the rectangle: ")
    fmt.Scan(&length)
    fmt.Print("Enter the width of the rectangle: ")
    fmt.Scan(&width)
    perimeter := 2 * (length + width)
    fmt.Printf("The perimeter of the rectangle is %d\n", perimeter)
}

9. Calculate the Volume of a Cube

Write a program that takes the side length of a cube, calculates the volume, and prints it.

Solution 
package main
import "fmt"

func main() {
    var sideLength int
    fmt.Print("Enter the side length of the cube: ")
    fmt.Scan(&sideLength)
    volume := sideLength * sideLength * sideLength
    fmt.Printf("The volume of the cube is %d\n", volume)
}

10. Convert Kilometers to Miles

Write a program that converts a distance from kilometers to miles.

Solution 
package main
import "fmt"

func main() {
    var kilometers float64
    fmt.Print("Enter distance in kilometers: ")
    fmt.Scan(&kilometers)
    miles := kilometers * 0.621371
    fmt.Printf("Distance in miles is %.2f\n", miles)
}
