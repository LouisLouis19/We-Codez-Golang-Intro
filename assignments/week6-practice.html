1. Find Even and Odd Numbers

Create a function separate_even_odd that takes a list of numbers and returns two lists: one containing all even numbers and the other containing all odd numbers.

Solution 
def separate_even_odd(numbers):
    even_numbers = []
    odd_numbers = []
    for num in numbers:
        if num % 2 == 0:
            even_numbers.append(num)
        else:
            odd_numbers.append(num)
    return even_numbers, odd_numbers

evens, odds = separate_even_odd([1, 2, 3, 4, 5, 6])
print("Even numbers:", evens)
print("Odd numbers:", odds)

2. Calculate Factorial with Error Handling

Write a function calculate_factorial that takes a non-negative integer and returns its factorial. Use error handling to manage cases where the input is not a valid non-negative integer.

Solution 
def calculate_factorial(n):
    try:
        if n < 0:
            raise ValueError("The number must be non-negative.")
        result = 1
        for i in range(1, n + 1):
            result *= i
        return result
    except ValueError as e:
        return str(e)

print(calculate_factorial(5))  # Output: 120
print(calculate_factorial(-3))  # Output: The number must be non-negative.

3. Find Prime Numbers in a Range

Create a function find_primes_in_range that takes two integers start and end and returns a list of all prime numbers between start and end (inclusive). Implement error handling for invalid inputs.

Solution 
def find_primes_in_range(start, end):
    try:
        if start < 2 or end < 2 or start > end:
            raise ValueError("Invalid range. Ensure that start and end are >= 2 and start <= end.")
        primes = []
        for num in range(start, end + 1):
            if num > 1:
                for i in range(2, int(num**0.5) + 1):
                    if num % i == 0:
                        break
                else:
                    primes.append(num)
        return primes
    except ValueError as e:
        return str(e)

print(find_primes_in_range(10, 20))  # Output: [11, 13, 17, 19]
print(find_primes_in_range(1, 20))   # Output: Invalid range. Ensure that start and end are >= 2 and start <= end.

4. Generate Fibonacci Sequence

Write a function generate_fibonacci that takes an integer n and returns the first n numbers in the Fibonacci sequence. Include error handling for cases where n is not a positive integer.

Solution 
def generate_fibonacci(n):
    try:
        if n <= 0:
            raise ValueError("The number must be a positive integer.")
        sequence = [0, 1]
        while len(sequence) < n:
            sequence.append(sequence[-1] + sequence[-2])
        return sequence[:n]
    except ValueError as e:
        return str(e)

print(generate_fibonacci(5))  # Output: [0, 1, 1, 2, 3]
print(generate_fibonacci(-3)) # Output: The number must be a positive integer.

5. Find Maximum and Minimum with Error Handling

Create a function find_max_min that takes a list of numbers and returns the maximum and minimum values in the list. Include error handling for cases where the list is empty or contains non-numeric values.

Solution 
def find_max_min(numbers):
    try:
        if not numbers:
            raise ValueError("The list is empty.")
        if any(not isinstance(num, (int, float)) for num in numbers):
            raise ValueError("The list contains non-numeric values.")
        return max(numbers), min(numbers)
    except ValueError as e:
        return str(e)

print(find_max_min([10, 5, 12, 8]))  # Output: (12, 5)
print(find_max_min([]))              # Output: The list is empty.
print(find_max_min([10, 'a', 5]))    # Output: The list contains non-numeric values.

6. Count Vowels and Consonants

Write a function count_vowels_consonants that takes a string and returns the number of vowels and consonants in the string. Ensure that the function handles empty strings and invalid inputs.

Solution 
def count_vowels_consonants(text):
    try:
        if not isinstance(text, str):
            raise ValueError("Input must be a string.")
        vowels = "aeiouAEIOU"
        num_vowels = sum(1 for char in text if char in vowels)
        num_consonants = sum(1 for char in text if char.isalpha() and char not in vowels)
        return num_vowels, num_consonants
    except ValueError as e:
        return str(e)

print(count_vowels_consonants("Hello World"))  # Output: (3, 7)
print(count_vowels_consonants(""))             # Output: (0, 0)
print(count_vowels_consonants(123))            # Output: Input must be a string.

7. Extract Even and Odd Indexed Elements

Create a function extract_even_odd_indexed that takes a list of numbers and returns two lists: one with elements at even indices and the other with elements at odd indices. Include error handling for empty lists.

Solution 
def extract_even_odd_indexed(numbers):
    try:
        if not numbers:
            raise ValueError("The list is empty.")
        even_indexed = [numbers[i] for i in range(0, len(numbers), 2)]
        odd_indexed = [numbers[i] for i in range(1, len(numbers), 2)]
        return even_indexed, odd_indexed
    except ValueError as e:
        return str(e)

evens, odds = extract_even_odd_indexed([1, 2, 3, 4, 5, 6])
print("Even indexed:", evens)
print("Odd indexed:", odds)
print(extract_even_odd_indexed([]))  # Output: The list is empty.

8. Sum of List with Default Step Parameter

Create a function sum_list that takes a list of numbers and an optional step parameter (default value is 1). The function should return the sum of numbers in the list, stepping through the list according to the given step.

Solution 
def sum_list(numbers, step=1):
    return sum(numbers[::step])

numbers = [1, 2, 3, 4, 5, 6]
print(sum_list(numbers))      # Output: 21
print(sum_list(numbers, 2))   # Output: 9
