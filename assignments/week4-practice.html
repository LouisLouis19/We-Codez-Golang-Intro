Sum of a List

Create a function called `sum_of_list` that takes a list of numbers as a parameter and returns the sum of all numbers in the list using a for loop.

Solution 
def sum_of_list(numbers):
    total = 0
    for num in numbers:
        total += num
    return total

print(sum_of_list([1, 2, 3, 4, 5]))  # Output: 15

Factorial of a Number

Create a function called `factorial` that takes an integer as a parameter and returns the factorial of that number using a while loop.

Solution 
def factorial(n):
    result = 1
    while n > 1:
        result *= n
        n -= 1
    return result

print(factorial(5))  # Output: 120

Count Vowels in a String

Create a function called `count_vowels` that takes a string as a parameter and returns the number of vowels (a, e, i, o, u) in the string.

Solution 
def count_vowels(s):
    vowels = "aeiouAEIOU"
    count = 0
    for char in s:
        if char in vowels:
            count += 1
    return count

print(count_vowels("Hello World"))  # Output: 3

Reverse a String

Create a function called `reverse_string` that takes a string as a parameter and returns the string reversed.

Solution 
def reverse_string(s):
    reversed_str = ""
    for char in s:
        reversed_str = char + reversed_str
    return reversed_str

print(reverse_string("Hello"))  # Output: "olleH"

Sum of Even Numbers in a Range

Create a function called `sum_of_evens` that takes two integers as parameters (start and end) and returns the sum of all even numbers in that range using a for loop.

Solution 
def sum_of_evens(start, end):
    total = 0
    for num in range(start, end + 1):
        if num % 2 == 0:
            total += num
    return total

print(sum_of_evens(1, 10))  # Output: 30

Find Common Elements in Two Lists

Create a function called `common_elements` that takes two lists as parameters and returns a list of common elements between the two lists.

Solution 
def common_elements(list1, list2):
    common = []
    for element in list1:
        if element in list2:
            common.append(element)
    return common

print(common_elements([1, 2, 3], [2, 3, 4]))  # Output: [2, 3]

Prime Number Checker

Create a function called `is_prime` that takes an integer as a parameter and returns `True` if the number is prime, and `False` otherwise.

Solution 
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

print(is_prime(11))  # Output: True
print(is_prime(10))  # Output: False

Palindrome Checker

Create a function called `is_palindrome` that takes a string as a parameter and returns `True` if the string is a palindrome (reads the same forward and backward), and `False` otherwise.

Solution 
def is_palindrome(s):
    return s == s[::-1]

print(is_palindrome("racecar"))  # Output: True
print(is_palindrome("hello"))    # Output: False

Remove Duplicates from a List

Create a function called `remove_duplicates` that takes a list as a parameter and returns a new list with duplicates removed, preserving the original order.

Solution 
def remove_duplicates(lst):
    unique = []
    for item in lst:
        if item not in unique:
            unique.append(item)
    return unique

print(remove_duplicates([1, 2, 2, 3, 4, 4, 5]))  # Output: [1, 2, 3, 4, 5]

Count Occurrences of an Element in a List

Create a function called `count_occurrences` that takes a list and an element as parameters and returns the number of occurrences of that element in the list.

Solution 
def count_occurrences(lst, element):
    count = 0
    for item in lst:
        if item == element:
            count += 1
    return count

print(count_occurrences([1, 2, 2, 3, 2], 2))  # Output: 3

Cumulative Sum of a List

Create a function called `cumulative_sum` that takes a list of numbers as a parameter and returns a list where each element is the cumulative sum of the numbers up to that index.

Solution 
def cumulative_sum(numbers):
    cumulative = []
    total = 0
    for num in numbers:
        total += num
        cumulative.append(total)
    return cumulative

print(cumulative_sum([1, 2, 3, 4]))  # Output: [1, 3, 6, 10]

Find the Second Largest Number in a List

Create a function called `second_largest` that takes a list of numbers as a parameter and returns the second largest number in the list.

Solution 
def second_largest(numbers):
    first, second = float('-inf'), float('-inf')
    for num in numbers:
        if num > first:
            second = first
            first = num
        elif num > second and num != first:
            second = num
    return second

print(second_largest([3, 5, 7, 2, 8]))  # Output: 7

Find the Index of an Element in a List

Create a function called `find_index` that takes a list and an element as parameters and returns the index of the element in the list. If the element is not found, return `-1`.

Solution 
def find_index(lst, element):
    for i in range(len(lst)):
        if lst[i] == element:
            return i
    return -1

print(find_index([10, 20, 30, 40], 30))  # Output: 2
print(find_index([10, 20, 30, 40], 50))  # Output: -1

Find the Longest Word in a List of Words

Create a function called `longest_word` that takes a list of words as a parameter and returns the longest word in the list. If there are multiple words with the same length, return the first one.

Solution 
def longest_word(words):
    longest = words[0]
    for word in words:
        if len(word) > len(longest):
            longest = word
    return longest

print(longest_word(["apple", "banana", "cherry", "watermelon"]))  # Output: "watermelon"

Find the Sum of Digits of an Integer

Create a function called `sum_of_digits` that takes an integer as a parameter and returns the sum of its digits.

Solution 
def sum_of_digits(n):
    total = 0
    while n > 0:
        total += n % 10
        n //= 10
    return total

print(sum_of_digits(1234))  # Output: 10

Count the Number of Words in a Sentence

Create a function called `word_count` that takes a string (sentence) as a parameter and returns the number of words in the sentence.

Solution 
def word_count(sentence):
    words = sentence.split()
    return len(words)

print(word_count("Hello world, how are you?"))  # Output: 5
