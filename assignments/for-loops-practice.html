Go Practice Exercises - For Loops

Complete the following exercises to strengthen your understanding of For Loops Concepts.

1. Generate Multiples of a Fixed Number

Write a program that prints the first 10 multiples of the number 5 using a `for` loop. 

number := 5
Solution 
package main
import "fmt"

func main() {
    number := 5
    for i := 1; i <= 10; i++ {
        fmt.Println(number * i)
    }
}

2. Sum of Even Numbers Up to 20

Write a program that calculates the sum of all even numbers between 1 and 20 using a `for` loop.

Solution 
package main
import "fmt"

func main() {
    sumEven := 0
    for i := 2; i <= 20; i += 2 {
        sumEven += i
    }
    fmt.Println("Sum of even numbers:", sumEven)
}

3. Fibonacci Sequence

Write a program that prints the first 10 numbers of the Fibonacci sequence using a `for` loop.

Solution 
package main
import "fmt"

func main() {
    a, b := 0, 1
    fmt.Println(a)
    for i := 0; i < 9; i++ {
        fmt.Println(b)
        a, b = b, a+b
    }
}

4. Count Numbers Greater Than 10

Write a program that counts and prints the number of elements in a list that are greater than 10. 

numbers := []int{4, 12, 7, 15, 9, 20, 3}
Solution 
package main
import "fmt"

func main() {
    numbers := []int{4, 12, 7, 15, 9, 20, 3}
    count := 0
    for _, num := range numbers {
        if num > 10 {
            count++
        }
    }
    fmt.Println("Count of numbers greater than 10:", count)
}

5. Find the Maximum Number in a List

Write a program that finds and prints the maximum number from the list [10, 25, 15, 30, 5].

Solution 
package main
import "fmt"

func main() {
    numbers := []int{10, 25, 15, 30, 5}
    maxNum := numbers[0]
    for _, num := range numbers {
        if num > maxNum {
            maxNum = num
        }
    }
    fmt.Println("The maximum number is:", maxNum)
}

7. Factorial of a Number

Write a program that calculates the factorial of a number using a `for` loop. 

number := 5
Solution 
package main
import "fmt"

func main() {
    number := 5
    factorial := 1
    for i := 1; i <= number; i++ {
        factorial *= i
    }
    fmt.Println("Factorial:", factorial)
}

8. Count Odd and Even Numbers

Write a program that counts the number of odd and even numbers in a list using a `for` loop. 

numbers := []int{3, 8, 12, 15, 7, 10, 21, 14}
Solution 
package main
import "fmt"

func main() {
    numbers := []int{3, 8, 12, 15, 7, 10, 21, 14}
    countOdd, countEven := 0, 0
    for _, num := range numbers {
        if num%2 == 0 {
            countEven++
        } else {
            countOdd++
        }
    }
    fmt.Println("Count of even numbers:", countEven)
    fmt.Println("Count of odd numbers:", countOdd)
}

10. Count Numbers Greater Than Their Previous

Write a program that counts how many times a number in a list is greater than the number that came before it. 

numbers := []int{5, 3, 8, 6, 10, 15, 7, 18}
Solution 
package main
import "fmt"

func main() {
    numbers := []int{5, 3, 8, 6, 10, 15, 7, 18}
    count := 0
    for i := 1; i < len(numbers); i++ {
        if numbers[i] > numbers[i-1] {
            count++
        }
    }
    fmt.Println("Count of numbers greater than their previous:", count)
}
